% SDPO (Self-Distillation) per-token reverse-KL diagram
% One policy pi_theta is forwarded twice (teacher forcing, no generation) over
% the SAME completion tokens y sampled from the student:
%   - Student pass: conditioned on the question only (green)
%   - Teacher pass: conditioned on question + a correct expert demo (yellow)
% Each pass emits a DENSE distribution over the full vocabulary at every
% completion position (red). Loss is the per-token reverse KL(student || teacher),
% pulling the question-only policy toward its demo-conditioned self.

% \minidist{name}{x}{y}{bar heights list}
% Draws a small full-vocabulary bar chart (5 bars) centered at (x,y).
\newcommand{\minidist}[4]{%
  \begin{scope}[shift={(#2,#3)}]
    \node[distframe, minimum width=0.8cm, minimum height=0.8cm] (#1) at (0,0.02) {};
    \foreach \h [count=\i from 0] in #4 {
      \fill[densered!75] (\i*0.135-0.30,-0.30) rectangle (\i*0.135-0.21,-0.30+\h);
    }
    \draw[black!45, line width=0.3pt] (-0.34,-0.30) -- (0.30,-0.30);
  \end{scope}%
}

% \minidisthl{name}{x}{y}{bar heights list}{peak index} -- highlighted mismatch:
% peak bar in deep red, the rest muted, deep-red box border.
\newcommand{\minidisthl}[5]{%
  \begin{scope}[shift={(#2,#3)}]
    \node[distframe, draw=hilite, line width=0.6pt, fill=hilite!6,
          minimum width=0.8cm, minimum height=0.8cm] (#1) at (0,0.02) {};
    \foreach \h [count=\j from 0] in #4 {
      \ifnum\j=#5\relax
        \fill[hilite] (\j*0.135-0.30,-0.30) rectangle (\j*0.135-0.21,-0.30+\h);
      \else
        \fill[densered!30] (\j*0.135-0.30,-0.30) rectangle (\j*0.135-0.21,-0.30+\h);
      \fi
    }
    \draw[black!45, line width=0.3pt] (-0.34,-0.30) -- (0.30,-0.30);
  \end{scope}%
}
