% RLHF feedback loop for LLMs.
% Training data -> Agent -> completions -> Reward model -> scalar reward -> Agent.
% Member of the "RL feedback loop" family (see rl_loop, rlvr_loop, tooluse_rl).
\documentclass{article}
\usepackage[margin=0.4cm, paperwidth=19cm, paperheight=13cm]{geometry}
\usepackage{tikz}
\usepackage{lmodern}
\renewcommand{\familydefault}{\sfdefault}
\usepackage{xcolor}
\usetikzlibrary{positioning, arrows.meta, calc}
\input{styles_rl_loop.tex}
\pagestyle{empty}

\begin{document}
\begin{tikzpicture}

% --- nodes (generous spacing so the grey captions have room) ---
\node[agent]  (agent)  at (0,0)     {Agent\\[4pt]{\LARGE$\pi_\theta(\cdot)$}};
\node[data]   (data)   at (-6.0,0)  {Training\\data};
\node[scorer] (reward) at (0.6,4.0) {Reward\\model};

% --- prompts: training data -> agent (edge label above, caption below, same x) ---
\draw[loop] (data.east) -- (agent.west);
\node[edgelbl] at (-3.225,0.6)  {$s_i$};
\node[sublbl]  at (-3.225,-0.7) {Prompts};

% --- completions: agent -> reward model, up the right side ---
% edge label above caption, both right of the vertical, centered on its midpoint
\draw[loop] (agent.east) -- (4.5,0) -- (4.5,4.0) -- (reward.east);
\node[edgelbl] at (4.95,2.4) {$a_i$};
\node[sublbl]  at (4.95,1.55) {Completions};

% --- scalar reward: reward model -> agent, straight down ---
% edge label and caption side by side at the drop midpoint
\draw[loop] (reward.south -| agent.north) -- (agent.north);
\node[edgelbl] at (0.42,2.05) {$r_i$};
\node[sublbl]  at (1.85,2.0)  {Scalar\\Reward};

% --- policy update caption (centered under the agent) ---
\node[note]   at (0,-1.75) {$\theta_{t+1} = \theta_t + \alpha\,\nabla_\theta J(\pi_\theta)$};
\node[sublbl] at (0,-2.45) {Policy Update};

\end{tikzpicture}
\end{document}
